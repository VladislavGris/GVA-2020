number function gorl(number a, number b)		; Функция сравнивает 2 числа и возвращает наибольшее
begin
if[a > b] then
{
return a
}
else
{
return b
}
end
number function equality(number a, number b)		; Функция определяет, равны ли 2 числа
begin
if[a = b] then
{
return 1						; Возвращает 1, если числа равны
}
else
{
return 0						; Возвращает 0, если числа не равны
}
end
number function pr(symbol a, symbol b)
begin
symbol e <- a
a <- b
b <- e
print(a)
return 0
end
main
begin
number a <- 1011b

{

{
